  <script>

    let originalValue = null;

    function press(num) {

      document.getElementById("display").value += num;

    }

    function clearDisplay() {

      document.getElementById("display").value = '';

      originalValue = null;

    }

    function backspace() {

      const display = document.getElementById("display");

      display.value = display.value.slice(0, -1);

    }

    function confirmNumpad() {

      const display = document.getElementById("display");

      const value = display.value.trim();

      if (value) {

        document.getElementById("display").value = value; // Zorg ervoor dat de waarde blijft staan

      }

      document.getElementById("naarLocatie").focus();

    }

    function getOriginalValue() {

      const current = parseInt(document.getElementById("display").value.trim());

      if (!isNaN(current)) {

        if (originalValue === null) {

          originalValue = current;

        }

        return originalValue;

      }

      return null;

    }

    function incrementDisplay() {

      const display = document.getElementById("display");

      const current = parseInt(display.value.trim());

      const base = getOriginalValue();

      if (!isNaN(current) && base !== null) {

        display.value = current + base;

      }

    }

    function decrementDisplay() {

      const display = document.getElementById("display");

      const current = parseInt(display.value.trim());

      const base = getOriginalValue();

      if (!isNaN(current) && base !== null) {

        const result = current - base;

        display.value = result >= 0 ? result : 0;

      }

    }

    document.getElementById("display").addEventListener("keydown", function(e) {

      if (e.key === "Enter") {

        e.preventDefault();

        confirmNumpad();

      }

    });

    const fields = ["vanLocatie", "ean"];

    fields.forEach((id, index) => {

      const input = document.getElementById(id);

      input.addEventListener("keydown", function (e) {

        if (e.key === "Enter") {

          e.preventDefault();

          if (index < fields.length - 1) {

            document.getElementById(fields[index + 1]).focus();

          } else {

            document.getElementById("display").focus();

          }

        }

      });

    });

    document.getElementById("productForm").addEventListener("submit", function(e) {

      e.preventDefault();

      alert("Formulier verzonden!");

    });
